# code.py — Web Serial demo (auto-detects board hardware)
# Pairs with the Adafruit Learning System Web Serial embed.
#
# Works out of the box on: Circuit Playground Express/Bluefruit, CLUE,
# QT Py ESP32-S2/S3 family (NEOPIXEL_POWER handled), FunHouse (DotStar +
# capacitive touch), Feathers with a single NeoPixel, and most other
# CircuitPython boards. If LEDs stay dark on a board with NEOPIXEL_POWER,
# flip the value=True to value=False (some boards are active-low).
#
# Protocol:
#   IN   RGB:r,g,b      set LED color (stops rainbow)
#   IN   RAINBOW        cycle hues
#   IN   STOP           stop cycling
#   IN   HELLO          reply with CAPS line
#   OUT  CAPS:LED=...,A=...,B=...   capability report
#   OUT  BTN:A:1 / BTN:A:0          button/touch A press / release
#   OUT  BTN:B:1 / BTN:B:0          button/touch B press / release

import sys
import time
import board
import digitalio
import supervisor


def _setup_led():
    power = getattr(board, "NEOPIXEL_POWER", None)
    if power is not None:
        p = digitalio.DigitalInOut(power)
        p.switch_to_output(value=True)
    np_pin = getattr(board, "NEOPIXEL", None)
    if np_pin is not None:
        try:
            import neopixel
            bid = getattr(board, "board_id", "")
            n = 10 if bid.startswith("circuitplayground") else 1
            return "neopixel", neopixel.NeoPixel(np_pin, n, brightness=0.3, auto_write=False)
        except ImportError:
            pass
    clk = getattr(board, "DOTSTAR_CLOCK", None)
    dat = getattr(board, "DOTSTAR_DATA", None)
    if clk is not None and dat is not None:
        try:
            import adafruit_dotstar
            n = 5 if getattr(board, "board_id", "") == "adafruit_funhouse" else 1
            return "dotstar", adafruit_dotstar.DotStar(clk, dat, n, brightness=0.3, auto_write=False)
        except ImportError:
            pass
    return None, None


led_kind, led = _setup_led()


def led_fill(rgb):
    if led is None:
        return
    led.fill(rgb)
    led.show()


class _DigBtn:
    def __init__(self, dio):
        self.dio = dio

    def pressed(self):
        return not self.dio.value


# Capacitive touch with per-pad baseline + hysteresis. touchio's default
# (baseline + 100, sampled once) is unreliable on FunHouse-style pads —
# it flickers near the threshold and drifts with humidity. We sample the
# untouched baseline at startup, then use a wide press band and a narrower
# release band so a touch latches cleanly.
TOUCH_PRESS_DELTA = 300
TOUCH_RELEASE_DELTA = 150


class _TouchBtn:
    def __init__(self, touch_in, baseline):
        self.t = touch_in
        self.press_at = baseline + TOUCH_PRESS_DELTA
        self.release_at = baseline + TOUCH_RELEASE_DELTA
        self.is_pressed = False

    def pressed(self):
        v = self.t.raw_value
        if self.is_pressed:
            if v < self.release_at:
                self.is_pressed = False
        else:
            if v > self.press_at:
                self.is_pressed = True
        return self.is_pressed


def _digital(name):
    pin = getattr(board, name, None)
    if pin is None:
        return None
    b = digitalio.DigitalInOut(pin)
    b.switch_to_input(pull=digitalio.Pull.UP)
    return _DigBtn(b), "digital"


def _touch(name):
    pin = getattr(board, name, None)
    if pin is None:
        return None
    try:
        import touchio
    except ImportError:
        return None
    t = touchio.TouchIn(pin)
    # Let the pin settle, then sample N times and use the min as baseline.
    time.sleep(0.05)
    samples = []
    for _ in range(15):
        samples.append(t.raw_value)
        time.sleep(0.004)
    return _TouchBtn(t, min(samples)), "touch"


def _first(*candidates):
    for fn in candidates:
        r = fn()
        if r is not None:
            return r
    return None, None


btn_a, kind_a = _first(
    lambda: _digital("BUTTON_A"),
    lambda: _digital("BUTTON"),
    lambda: _digital("BOOT0"),
    lambda: _touch("CAP6"),
    lambda: _touch("BUTTON_DOWN"),
)
btn_b, kind_b = _first(
    lambda: _digital("BUTTON_B"),
    lambda: _touch("CAP7"),
    lambda: _touch("BUTTON_SELECT"),
)


def report_caps():
    print("CAPS:LED=%s,A=%s,B=%s" % (led_kind or "none",
                                     kind_a or "none",
                                     kind_b or "none"))


def _hsv(h):
    h = h % 1.0
    i = int(h * 6); f = h * 6 - i
    q = int(255 * (1 - f)); t = int(255 * f); i %= 6
    return [(255, t, 0), (q, 255, 0), (0, 255, t),
            (0, q, 255), (t, 0, 255), (255, 0, q)][i]


state = {"rainbow": False, "hue": 0.0}
last_a = btn_a.pressed() if btn_a else False
last_b = btn_b.pressed() if btn_b else False
buf = ""

while True:
    n = supervisor.runtime.serial_bytes_available
    while n:
        ch = sys.stdin.read(1)
        n -= 1
        if ch in ("\n", "\r"):
            line = buf.strip(); buf = ""
            if not line:
                continue
            if line.startswith("RGB:") and led is not None:
                try:
                    r, g, b = [int(x) for x in line[4:].split(",")]
                    state["rainbow"] = False
                    led_fill((r, g, b))
                except ValueError:
                    pass
            elif line == "RAINBOW":
                state["rainbow"] = True
            elif line == "STOP":
                state["rainbow"] = False
            elif line == "HELLO":
                report_caps()
        else:
            buf += ch

    if state["rainbow"] and led is not None:
        state["hue"] = (state["hue"] + 0.005) % 1.0
        led_fill(_hsv(state["hue"]))

    if btn_a is not None:
        cur = btn_a.pressed()
        if cur != last_a:
            print("BTN:A:" + ("1" if cur else "0"))
            last_a = cur
    if btn_b is not None:
        cur = btn_b.pressed()
        if cur != last_b:
            print("BTN:B:" + ("1" if cur else "0"))
            last_b = cur

    time.sleep(0.01)
